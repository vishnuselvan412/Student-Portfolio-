// ============================================================ // view.js — Portfolio View Page (PUBLIC) // Anyone with ?uid=xxx can view. No login required to view. // ============================================================ import { auth, db } from "./firebase.js"; import { onAuthStateChanged } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-auth.js"; import { doc, getDoc, collection, getDocs } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-firestore.js"; // ─── Init ───────────────────────────────────────────────────── // Portfolio is PUBLIC — anyone with ?uid=xxx can view it. // If no uid in URL, fall back to the currently logged-in user. async function init() { const params = new URLSearchParams(window.location.search); const uidFromUrl = params.get("uid"); if (uidFromUrl) { // Public view — no login required await renderPortfolio(uidFromUrl); hideLoader(); setupScrollObserver(); setupNav(); } else { // No uid in URL — wait for auth, then redirect to shareable URL onAuthStateChanged(auth, async (user) => { if (!user) { window.location.href = "index.html"; return; } // Rewrite URL so it becomes shareable window.history.replaceState({}, "", `view.html?uid=${user.uid}`); await renderPortfolio(user.uid); hideLoader(); setupScrollObserver(); setupNav(); }); } } init(); // ─── Render All ─────────────────────────────────────────────── async function renderPortfolio(uid) { try { const [profileSnap, linksSnap, eduSnap, projectsSnap, resumesSnap, certsSnap] = await Promise.all([ getDoc(doc(db, "users", uid, "data", "profile")), getDoc(doc(db, "users", uid, "data", "links")), getDoc(doc(db, "users", uid, "data", "education")), getDocs(collection(db, "users", uid, "projects")), getDocs(collection(db, "users", uid, "resumes")), getDocs(collection(db, "users", uid, "certificates")), ]); renderProfile( profileSnap.exists() ? profileSnap.data() : {}, linksSnap.exists() ? linksSnap.data() : {} ); renderEducation(eduSnap.exists() ? eduSnap.data() : {}); const projects = []; projectsSnap.forEach(d => projects.push(d.data())); renderProjects(projects); const resumes = []; resumesSnap.forEach(d => resumes.push(d.data())); renderFiles(resumes, "resume-grid", "resume-section"); const certs = []; certsSnap.forEach(d => certs.push(d.data())); renderFiles(certs, "cert-grid", "cert-section"); } catch (err) { console.error("Error loading portfolio:", err); } } // ─── Profile & Hero ─────────────────────────────────────────── function renderProfile(profile, links) { const name = profile.fullName || "Portfolio"; const bio = profile.bio || "Welcome to my portfolio."; const dept = profile.department || "Student"; // Split name — accent last word const nameParts = name.trim().split(" "); const firstName = nameParts.slice(0, -1).join(" "); const lastName = nameParts.slice(-1)[0]; document.getElementById("hero-name").innerHTML = firstName ? `${firstName} ${lastName}` : `${lastName}`; document.getElementById("hero-bio").textContent = bio; document.getElementById("hero-department").textContent = dept; document.getElementById("footer-name").textContent = name; document.title = `${name} — Portfolio`; // Meta const metaEl = document.getElementById("hero-meta"); const metaItems = []; if (profile.email) metaItems.push({ icon: "✉", text: profile.email }); if (profile.phone) metaItems.push({ icon: "✆", text: profile.phone }); if (dept) metaItems.push({ icon: "◈", text: dept }); metaEl.innerHTML = metaItems.map(m => `
${m.icon} ${m.text}
`).join(""); // Social Links const linksEl = document.getElementById("hero-links"); const linkDefs = [ { label: "GitHub", icon: "⌥", url: links.github }, { label: "LinkedIn", icon: "in", url: links.linkedin }, { label: "LeetCode", icon: "{ }", url: links.leetcode }, ]; linksEl.innerHTML = linkDefs .filter(l => l.url) .map(l => ` ${l.icon} ${l.label} `).join(""); } // ─── Education ──────────────────────────────────────────────── function renderEducation(ed) { const timeline = document.getElementById("edu-timeline"); const items = []; if (ed.school10?.name) { items.push({ type: "school", badge: "10th Grade", institution: ed.school10.name, degree: ed.school10.board || "", tags: [ ed.school10.year && `📅 ${ed.school10.year}`, ed.school10.grade && `Grade: ${ed.school10.grade}`, ].filter(Boolean) }); } if (ed.school12?.name) { items.push({ type: "school", badge: "12th Grade", institution: ed.school12.name, degree: `${ed.school12.board || ""} ${ed.school12.stream ? "· " + ed.school12.stream : ""}`, tags: [ ed.school12.year && `📅 ${ed.school12.year}`, ed.school12.grade && `Grade: ${ed.school12.grade}`, ].filter(Boolean) }); } if (ed.ug?.collegeName) { items.push({ type: "college", badge: ed.ug.degree || "Undergraduate", institution: ed.ug.collegeName, degree: `${ed.ug.branch || ""} ${ed.ug.startYear ? "· " + ed.ug.startYear : ""} – ${ed.ug.endYear || "Present"}`, tags: [ ed.ug.grade && `CGPA / %: ${ed.ug.grade}`, ].filter(Boolean) }); } if (ed.pg?.collegeName) { items.push({ type: "college", badge: ed.pg.degree || "Postgraduate", institution: ed.pg.collegeName, degree: `${ed.pg.branch || ""} ${ed.pg.startYear ? "· " + ed.pg.startYear : ""} – ${ed.pg.endYear || "Present"}`, tags: [ ed.pg.grade && `CGPA / %: ${ed.pg.grade}`, ].filter(Boolean) }); } if (items.length === 0) { document.getElementById("edu-section").style.display = "none"; return; } timeline.innerHTML = items.map(item => `
${item.type === "school" ? "🏫" : "🎓"}
${item.badge}
${item.institution}
${item.degree}
${item.tags.map(t => `${t}` ).join("")}
`).join(""); } // ─── Projects ───────────────────────────────────────────────── function renderProjects(projects) { const bento = document.getElementById("projects-bento"); if (projects.length === 0) { document.getElementById("proj-section").style.display = "none"; return; } bento.innerHTML = projects.map((proj, i) => `
${String(i + 1).padStart(2, "0")}

${proj.title}

${proj.description || "No description provided."}

${proj.techStack ? `
${proj.techStack.split(",").map(t => `${t.trim()}`).join("")}
` : ""}
${proj.github ? `⌥ GitHub` : ""} ${proj.demo ? `↗ Live Demo` : ""}
`).join(""); } // ─── Files (Resumes & Certificates) ────────────────────────── function renderFiles(files, gridId, sectionId) { const grid = document.getElementById(gridId); if (files.length === 0) { document.getElementById(sectionId).style.display = "none"; return; } grid.innerHTML = files.map(file => { const sizeKB = (file.size / 1024).toFixed(1); const date = file.uploadedAt?.toDate ? file.uploadedAt.toDate().toLocaleDateString() : "—"; let iconClass = "doc", iconEmoji = "📄"; if (file.type?.includes("pdf")) { iconClass = "pdf"; iconEmoji = "📕"; } else if (file.type?.includes("image")) { iconClass = "img"; iconEmoji = "🖼"; } return `
${iconEmoji}
${file.name}
${sizeKB} KB · ${date}
${file.url || file.downloadURL ? `↓ View / Download` : `Not available`}
`; }).join(""); } // ─── Scroll Observer ────────────────────────────────────────── function setupScrollObserver() { const observer = new IntersectionObserver( entries => entries.forEach(e => { if (e.isIntersecting) e.target.classList.add("visible"); }), { threshold: 0.1 } ); document.querySelectorAll(".port-section").forEach(s => observer.observe(s)); } // ─── Nav ────────────────────────────────────────────────────── function setupNav() { const nav = document.getElementById("view-nav"); window.addEventListener("scroll", () => { nav.classList.toggle("scrolled", window.scrollY > 60); }); // Share button — copies current URL (which already has ?uid=xxx) document.getElementById("share-btn").addEventListener("click", () => { navigator.clipboard.writeText(window.location.href).then(() => { const toast = document.getElementById("toast"); toast.classList.add("show"); setTimeout(() => toast.classList.remove("show"), 2500); }); }); } // ─── Loader ─────────────────────────────────────────────────── function hideLoader() { setTimeout(() => document.getElementById("loader-screen").classList.add("hidden"), 400); }
